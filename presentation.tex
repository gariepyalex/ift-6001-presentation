\begin{frame}
  \frametitle{Saisie directe}
  \begin{itemize}
  \item Prédiction directe d'un point de saisie
  \item 6 sorties: $[x, y, w, h, 2\sin(\theta), 2\cos(\theta)]$
  \item Suppression de symétries en utilisant $(2\sin(\theta), 2\cos(\theta))$
  \item \emph{loss} L2: $\sqrt{(x_{réel} - x_{prédit})^2 + (y_{réel} - y_{prédit})^2
    + ...}$
  \end{itemize}
  
\end{frame}
